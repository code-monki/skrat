287/**
288 * @copydoc PdfGraphicsView::rebuildPages
289 */
290void PdfGraphicsView::rebuildPages()
291{
292 clearScene();
293 if (!m_document || m_document->pageCount() <= 0) {
294 return;
295 }
296
297 // Keep visible gutters around pages so fit-width mode doesn't look clipped.
298 const qreal topPad = 24.0;
299 const qreal pageMatPx = 14.0;
300 const qreal fitWidthGutterPx = 0.0;
301 const qreal fitOuterPad = pageMatPx + fitWidthGutterPx;
302 const qreal leftPad = (m_zoomMode == ZoomMode::Custom) ? 56.0 : pageMatPx;
303 const bool fitMode = (m_zoomMode == ZoomMode::FitToWidth || m_zoomMode == ZoomMode::FitInView);
304 qreal y = topPad;
305 qreal maxWidth = 0.0;
306
307 const int pageCount = m_document->pageCount();
308 m_pages.reserve(pageCount);
309
310 for (int page = 0; page < pageCount; ++page) {
311 const QSizeF pagePt = m_document->pagePointSize(page);
312 if (pagePt.isEmpty()) {
313 continue;
314 }
315 qreal scale = m_zoomFactor;
316 if (m_zoomMode == ZoomMode::FitToWidth || m_zoomMode == ZoomMode::FitInView) {
317 const qreal usableWidth = qMax<qreal>(
318 80.0,
319 viewport()->width() - (2.0 * fitOuterPad));
320 const qreal widthScale = qMax<qreal>(0.05, usableWidth / pagePt.width());
321 if (m_zoomMode == ZoomMode::FitToWidth) {
322 scale = widthScale;
323 } else {
324 const qreal usableHeight = qMax<qreal>(80.0, viewport()->height() - 2.0 * (topPad + pageMatPx));
325 const qreal heightScale = qMax<qreal>(0.05, usableHeight / pagePt.height());
326 scale = qMin(widthScale, heightScale);
327 }
328 }
329 const QSize pagePx(qMax(1, qRound(pagePt.width() * scale)),
330 qMax(1, qRound(pagePt.height() * scale)));
331 const QImage img = m_document->render(page, pagePx);
332 if (img.isNull()) {
333 continue;
334 }
335
336 // Draw a white page mat so gutters are visually calm in dark UI chrome.
337 auto *pageMat = m_scene->addRect(
338 QRectF(QPointF(leftPad - pageMatPx, y - pageMatPx),
339 QSizeF(pagePx.width() + pageMatPx * 2.0, pagePx.height() + pageMatPx * 2.0)),
340 Qt::NoPen,
341 QBrush(QColor(255, 255, 255)));
342 pageMat->setZValue(0.25);
343
344 // Draw a subtle page edge so page bounds are visible.
345 auto *pageFrame = m_scene->addRect(QRectF(QPointF(leftPad, y), QSizeF(pagePx)),
346 QPen(QColor(155, 155, 155, 165), 1.0),
347 Qt::NoBrush);
348 pageFrame->setZValue(0.5);
349
350 auto *item = m_scene->addPixmap(QPixmap::fromImage(img));
351 item->setTransformationMode(Qt::SmoothTransformation);
352 item->setPos(leftPad, y);
353 item->setZValue(1.0);
354
355 PageItem pageItem;
356 pageItem.pixmapItem = item;
357 pageItem.sceneRect = QRectF(QPointF(leftPad, y), QSizeF(pagePx));
358 pageItem.pageSizePt = pagePt;
359 pageItem.scale = scale;
360 m_pages.push_back(pageItem);
361
362 y += pagePx.height() + m_pageSpacingPx;
363 maxWidth = qMax(maxWidth, static_cast<qreal>(pagePx.width()));
364 }
365
366 const qreal contentWidth = maxWidth + leftPad * 2.0;
367 const qreal sceneWidth = fitMode ? qMax(contentWidth, static_cast<qreal>(viewport()->width()))
368 : contentWidth;
369 m_scene->setSceneRect(0.0, 0.0, sceneWidth, y + topPad);
370 if (fitMode) {
371 horizontalScrollBar()->setValue(horizontalScrollBar()->minimum());
372 }
373 rebuildTextSelectionHighlights();
374}

376/**
377 * @copydoc PdfGraphicsView::rebuildSearchHighlights
378 */
379void PdfGraphicsView::rebuildSearchHighlights()
380{
381 for (auto *item : m_searchHighlights) {
382 if (item) {
383 m_scene->removeItem(item);
384 delete item;
385 }
386 }
387 m_searchHighlights.clear();
388
389 if (!m_searchModel || !m_document || m_pages.isEmpty()) {
390 return;
391 }
392
393 const int total = m_searchModel->rowCount(QModelIndex());
394 for (int i = 0; i < total; ++i) {
395 const QPdfLink link = m_searchModel->resultAtIndex(i);
396 if (!link.isValid()) {
397 continue;
398 }
399 const QList<QRectF> rects = link.rectangles();
400 if (!rects.isEmpty()) {
401 for (const QRectF &r : rects) {
402 const QRectF sr = mapPdfRectToSceneRect(link.page(), r);
403 if (!sr.isValid()) {
404 continue;
405 }
406 auto *hl = m_scene->addRect(
407 sr,
408 Qt::NoPen,
409 QBrush(i == m_currentSearchResultIndex ? QColor(255, 170, 0, 120)
410 : QColor(255, 235, 120, 90)));
411 hl->setZValue(10.0);
412 m_searchHighlights.push_back(hl);
413 }
414 } else {
415 const QPointF pt = mapPdfPointToScenePoint(link.page(), link.location());
416 const QSizeF approx(14.0, 8.0);
417 auto *hl = m_scene->addRect(QRectF(pt, approx),
418 Qt::NoPen,
419 QBrush(i == m_currentSearchResultIndex ? QColor(255, 170, 0, 120)
420 : QColor(255, 235, 120, 90)));
421 hl->setZValue(10.0);
422 m_searchHighlights.push_back(hl);
423 }
424 }
425}

427/**
428 * @copydoc PdfGraphicsView::rebuildTextSelectionHighlights
429 */
430void PdfGraphicsView::rebuildTextSelectionHighlights()
431{
432 for (auto *item : m_textSelectionHighlights) {
433 if (item) {
434 m_scene->removeItem(item);
435 delete item;
436 }
437 }
438 m_textSelectionHighlights.clear();
439 if (!m_document || !m_hasTextSelection || m_selectionAnchorPage < 0 || m_selectionCurrentPage < 0) {
440 return;
441 }
442 syncDocSelectionCharRange();
443 const bool forward = selectionAnchorComesBeforeCurrent();
444 const int startPage = forward ? m_selectionAnchorPage : m_selectionCurrentPage;
445 const int endPage = forward ? m_selectionCurrentPage : m_selectionAnchorPage;
446 const QPointF startPoint = forward ? m_selectionAnchorPdf : m_selectionCurrentPdf;
447 const QPointF endPoint = forward ? m_selectionCurrentPdf : m_selectionAnchorPdf;
448
449 for (int page = startPage; page <= endPage; ++page) {
450 if (page < 0 || page >= m_pages.size()) {
451 continue;
452 }
453 const QPdfSelection selection =
454 pdfRangeForPageInSelection(page, page == startPage, page == endPage, startPoint, endPoint);
455 const QList<QPolygonF> bounds = selection.bounds();
456 bool drew = false;
457 for (const QPolygonF &poly : bounds) {
458 if (poly.isEmpty()) {
459 continue;
460 }
461 QPolygonF scenePoly;
462 scenePoly.reserve(poly.size());
463 for (const QPointF &pt : poly) {
464 scenePoly << mapPdfPointToScenePoint(page, pt);
465 }
466 auto *hl = m_scene->addPolygon(scenePoly, Qt::NoPen, QBrush(QColor(64, 148, 255, 90)));
467 hl->setZValue(9.0);
468 m_textSelectionHighlights.push_back(hl);
469 drew = true;
470 }
471 // Some PDFs return text but no bounds; still show a band so cross-page drags don't
472 // look like the selection vanished.
473 if (!drew && !selection.text().trimmed().isEmpty()) {
474 const QSizeF psz = m_pages[page].pageSizePt;
475 QRectF pdfBand;
476 if (page == startPage && page == endPage) {
477 const qreal y1 = qMin(startPoint.y(), endPoint.y());
478 const qreal y2 = qMax(startPoint.y(), endPoint.y());
479 pdfBand = QRectF(0.0, y1, psz.width(), qMax<qreal>(0.0, y2 - y1));
480 } else if (page == startPage) {
481 pdfBand = QRectF(0.0, startPoint.y(), psz.width(), qMax<qreal>(0.0, psz.height() - startPoint.y()));
482 } else if (page == endPage) {
483 pdfBand = QRectF(0.0, 0.0, psz.width(), qMax<qreal>(0.0, endPoint.y()));
484 } else {
485 pdfBand = QRectF(0.0, 0.0, psz.width(), psz.height());
486 }
487 const QRectF sr = mapPdfRectToSceneRect(page, pdfBand);
488 if (sr.isValid() && sr.width() > 1.0 && sr.height() > 1.0) {
489 QPolygonF band;
490 band << sr.topLeft() << sr.topRight() << sr.bottomRight() << sr.bottomLeft();
491 auto *hl = m_scene->addPolygon(band, Qt::NoPen, QBrush(QColor(64, 148, 255, 90)));
492 hl->setZValue(9.0);
493 m_textSelectionHighlights.push_back(hl);
494 }
495 }
496 }
497}

539/**
540 * @copydoc PdfGraphicsView::scenePointToPdfPoint
541 */
542bool PdfGraphicsView::scenePointToPdfPoint(const QPointF &scenePoint, int *page, QPointF *pdfPoint) const
543{
544 if (!page || !pdfPoint) {
545 return false;
546 }
547 if (m_pages.isEmpty()) {
548 return false;
549 }
550 for (int i = 0; i < m_pages.size(); ++i) {
551 if (!m_pages[i].sceneRect.contains(scenePoint)) {
552 continue;
553 }
554 *page = i;
555 *pdfPoint = clampPdfPointToPage(i, mapScenePointToPdfPoint(i, scenePoint));
556 return true;
557 }
558 const qreal y = scenePoint.y();
559 // Above first or below last page: keep continuous mapping (clamp), not a random "nearest" page.
560 if (y < m_pages.first().sceneRect.top()) {
561 *page = 0;
562 *pdfPoint = clampPdfPointToPage(0, mapScenePointToPdfPoint(0, scenePoint));
563 return true;
564 }
565 if (y > m_pages.last().sceneRect.bottom()) {
566 const int li = m_pages.size() - 1;
567 *page = li;
568 *pdfPoint = clampPdfPointToPage(li, mapScenePointToPdfPoint(li, scenePoint));
569 return true;
570 }
571 // Inter-page vertical gap: pick upper vs lower page by midpoint so the caret does not
572 // jump to top/bottom edges and lose text for one frame.
573 for (int i = 0; i < m_pages.size() - 1; ++i) {
574 const qreal gapTop = m_pages[i].sceneRect.bottom();
575 const qreal gapBot = m_pages[i + 1].sceneRect.top();
576 if (y > gapTop && y < gapBot) {
577 const qreal mid = (gapTop + gapBot) * 0.5;
578 if (y < mid) {
579 *page = i;
580 *pdfPoint = clampPdfPointToPage(i, mapScenePointToPdfPoint(i, scenePoint));
581 } else {
582 *page = i + 1;
583 *pdfPoint = clampPdfPointToPage(i + 1, mapScenePointToPdfPoint(i + 1, scenePoint));
584 }
585 return true;
586 }
587 }
588 // Wider scene than a page: y overlaps a page band but x is in the left/right mat — pick
589 // the row-overlapping page whose horizontal range is closest to the cursor.
590 int bestPage = -1;
591 qreal bestHoriz = std::numeric_limits<qreal>::max();
592 for (int i = 0; i < m_pages.size(); ++i) {
593 const QRectF r = m_pages[i].sceneRect;
594 if (y < r.top() || y > r.bottom()) {
595 continue;
596 }
597 const qreal d = (scenePoint.x() < r.left()) ? (r.left() - scenePoint.x())
598 : (scenePoint.x() > r.right()) ? (scenePoint.x() - r.right())
599 : 0.0;
600 if (d < bestHoriz) {
601 bestHoriz = d;
602 bestPage = i;
603 }
604 }
605 if (bestPage >= 0) {
606 *page = bestPage;
607 *pdfPoint = clampPdfPointToPage(bestPage, mapScenePointToPdfPoint(bestPage, scenePoint));
608 return true;
609 }
610 // Last resort: page whose scene rect is closest in Y to the point (legacy behavior).
611 int nearest = 0;
612 qreal bestDistY = std::numeric_limits<qreal>::max();
613 for (int i = 0; i < m_pages.size(); ++i) {
614 const QRectF r = m_pages[i].sceneRect;
615 const qreal clampedY = qBound(r.top(), y, r.bottom());
616 const qreal dist = qAbs(y - clampedY);
617 if (dist < bestDistY) {
618 bestDistY = dist;
619 nearest = i;
620 }
621 }
622 *page = nearest;
623 *pdfPoint = clampPdfPointToPage(nearest, mapScenePointToPdfPoint(nearest, scenePoint));
624 return true;
625}

643bool PdfGraphicsView::selectionRangeHasText()
644{

655 const QPointF sp = forward ? m_selectionAnchorPdf : m_selectionCurrentPdf;
656 const QPointF ep = forward ? m_selectionCurrentPdf : m_selectionAnchorPdf;

669/**
670 * @copydoc PdfGraphicsView::syncDocSelectionCharRange
671 */
672void PdfGraphicsView::syncDocSelectionCharRange()
673{
674 m_selDocFromPage = -1;
675 m_selDocFromChar = -1;
676 m_selDocToPage = -1;
677 m_selDocToChar = -1;
678 if (m_selectionAnchorPage < 0 || m_selectionCurrentPage < 0) {
679 return;
680 }
681 if (m_selectionAnchorChar < 0 || m_selectionCurrentChar < 0) {
682 return;
683 }
684 if (selectionAnchorComesBeforeCurrentCharOrder()) {
685 m_selDocFromPage = m_selectionAnchorPage;

711int PdfGraphicsView::pdfCharStartIndexAtPoint(int page, const QPointF &pdfPt) const
712{
713 if (!m_document || page < 0 || page >= m_pages.size()) {
714 return -1;
715 }
716 const QSizeF sz = m_pages[page].pageSizePt;
717 const QPointF p(qBound(0.0, pdfPt.x(), sz.width()), qBound(0.0, pdfPt.y(), sz.height()));
718 static constexpr qreal kProbeDx[] = { 1.0, 3.0, 10.0, 28.0, 80.0 };
719 for (const qreal dx : kProbeDx) {
720 const QPointF q(qMin(p.x() + dx, sz.width()), p.y());
721 const QPdfSelection s = m_document->getSelection(page, p, q);
722 if (s.startIndex() < s.endIndex()) {
723 return s.startIndex();
724 }
725 }
726 const qreal y = p.y();
727 const QPdfSelection line = m_document->getSelection(
728 page, QPointF(0.0, y), QPointF(sz.width(), qMin(y + 2.0, sz.height())));
729 if (line.startIndex() < line.endIndex()) {
730 return line.startIndex();
731 }
732 return -1;

736 * @copydoc PdfGraphicsView::pdfCharInclusiveEndIndexAtPoint

738int PdfGraphicsView::pdfCharInclusiveEndIndexAtPoint(int page, const QPointF &pdfPt) const
739{
740 if (!m_document || page < 0 || page >= m_pages.size()) {
741 return -1;
742 }
743 const QSizeF sz = m_pages[page].pageSizePt;
744 const QPointF p(qBound(0.0, pdfPt.x(), sz.width()), qBound(0.0, pdfPt.y(), sz.height()));
745 // Tight probes first so a click without drag does not jump to the whole line.
746 static constexpr qreal kProbeDx[] = { 1.0, 3.0, 10.0, 28.0, 80.0 };
747 for (const qreal dx : kProbeDx) {
748 const QPointF q(qMin(p.x() + dx, sz.width()), p.y());
749 const QPdfSelection s = m_document->getSelection(page, p, q);
750 if (s.startIndex() < s.endIndex()) {
751 const int endExclusive = s.endIndex();
752 return qMax(s.startIndex(), endExclusive - 1);
753 }
754 }
755 const qreal y0 = p.y();
756 const qreal y1 = qMin(y0 + 6.0, sz.height());
757 const QPdfSelection line =
758 m_document->getSelection(page, QPointF(0.0, y0), QPointF(sz.width(), y1));
759 if (line.startIndex() < line.endIndex()) {
760 const int endExclusive = line.endIndex();
761 return qMax(line.startIndex(), endExclusive - 1);
762 }
763 return -1;

766/**
767 * @copydoc PdfGraphicsView::pdfRangeForPageInSelection
768 */
769QPdfSelection PdfGraphicsView::pdfRangeForPageInSelection(int page,
770 bool isStartPage,
771 bool isEndPage,
772 const QPointF &startPoint,
773 const QPointF &endPoint) const
774{
775 Q_ASSERT(m_document);
776 if (page < 0 || page >= m_pages.size()) {
777 return m_document->getSelection(0, QPointF(0.0, 0.0), QPointF(0.0, 0.0));
778 }
779 // Index-based range: QPdfDocument::getSelection often returns nothing when both corners
780 // resolve to the same character (Qt source: endIndex == startIndex). Character spans
781 // from getSelectionAtIndex match PDFium and behave correctly across page boundaries.
782 if (m_selDocFromPage >= 0 && m_selDocToPage >= 0 && m_selDocFromChar >= 0 && m_selDocToChar >= 0
783 && page >= m_selDocFromPage && page <= m_selDocToPage) {
784 const QPdfSelection all = m_document->getAllText(page);
785 const int allExclusiveEnd = all.endIndex();
786 if (allExclusiveEnd > 0) {
787 if (m_selDocFromPage == m_selDocToPage) {
788 const int a = qMin(m_selDocFromChar, m_selDocToChar);
789 const int b = qMax(m_selDocFromChar, m_selDocToChar);
790 return m_document->getSelectionAtIndex(page, a, b - a + 1);
791 }
792 if (page == m_selDocFromPage) {
793 return m_document->getSelectionAtIndex(page, m_selDocFromChar, allExclusiveEnd - m_selDocFromChar);
794 }
795 if (page == m_selDocToPage) {
796 return m_document->getSelectionAtIndex(page, 0, m_selDocToChar + 1);
797 }
798 return all;
799 }
800 }
801 if (isStartPage && isEndPage) {
802 const QSizeF pageSize = m_pages[page].pageSizePt;
803 const qreal dx = qAbs(endPoint.x() - startPoint.x());
804 const qreal dy = qAbs(endPoint.y() - startPoint.y());
805 if (dy > qMax<qreal>(4.0, dx * 1.15)) {
806 const qreal y1 = qMin(startPoint.y(), endPoint.y());
807 const qreal y2 = qMax(startPoint.y(), endPoint.y());
808 return m_document->getSelection(
809 page,
810 QPointF(0.0, y1),
811 QPointF(pageSize.width(), y2));
812 }
813 return m_document->getSelection(page, startPoint, endPoint);
814 }
815 const QPdfSelection allText = m_document->getAllText(page);
816 if (isStartPage) {
817 const QSizeF pageSize = m_pages[page].pageSizePt;
818 return m_document->getSelection(
819 page, QPointF(0.0, startPoint.y()), QPointF(pageSize.width(), pageSize.height()));
820 }
821 if (isEndPage) {
822 const QSizeF sz = m_pages[page].pageSizePt;
823 auto rectSel = [&](const QPointF &a, const QPointF &b) -> QPdfSelection {
824 const QPointF r1(qMin(a.x(), b.x()), qMin(a.y(), b.y()));
825 const QPointF r2(qMax(a.x(), b.x()), qMax(a.y(), b.y()));
826 return m_document->getSelection(page, r1, r2);
827 };
828 QPdfSelection sel = rectSel(QPointF(0.0, 0.0), QPointF(sz.width(), endPoint.y()));
829 if (!sel.text().trimmed().isEmpty()) {
830 return sel;
831 }
832 sel = rectSel(QPointF(0.0, 0.0), endPoint);
833 if (!sel.text().trimmed().isEmpty()) {
834 return sel;
835 }
836 sel = rectSel(QPointF(0.0, endPoint.y()), endPoint);
837 if (!sel.text().trimmed().isEmpty()) {
838 return sel;
839 }
840 return rectSel(QPointF(0.0, 0.0), endPoint);
841 }
842 return allText;
843}

880/**
881 * @copydoc PdfGraphicsView::updateTextSelection
882 */
883void PdfGraphicsView::updateTextSelection(int page, const QPointF &pdfPoint)
884{
885 const int candidateCurrentPage = page;
886 const QPointF candidateCurrentPdf = clampPdfPointToPage(page, pdfPoint);
887 // Always move the caret with the pointer while dragging. A previous early-return path
888 // skipped updating these when a transient range had no text, which broke cross-page
889 // highlights and copy (caret stayed on the previous page).
890 m_selectionCurrentPage = candidateCurrentPage;
891 m_selectionCurrentPdf = candidateCurrentPdf;
892 m_selectionCurrentChar = pdfCharInclusiveEndIndexAtPoint(candidateCurrentPage, candidateCurrentPdf);
893 syncDocSelectionCharRange();
894
895 bool hasAnyText = false;
896 const bool forward =
897 (m_selectionAnchorPage < candidateCurrentPage)
898 || (m_selectionAnchorPage == candidateCurrentPage
899 && (m_selectionAnchorPdf.y() < candidateCurrentPdf.y()
900 || (qFuzzyCompare(1.0 + m_selectionAnchorPdf.y(), 1.0 + candidateCurrentPdf.y())
901 && m_selectionAnchorPdf.x() <= candidateCurrentPdf.x())));
902 const int startPage = forward ? m_selectionAnchorPage : candidateCurrentPage;
903 const int endPage = forward ? candidateCurrentPage : m_selectionAnchorPage;
904 const QPointF startPoint = forward ? m_selectionAnchorPdf : candidateCurrentPdf;
905 const QPointF endPoint = forward ? candidateCurrentPdf : m_selectionAnchorPdf;
906
907 for (int p = startPage; p <= endPage; ++p) {
908 if (p < 0 || p >= m_pages.size()) {
909 continue;
910 }
911 const QPdfSelection selection =
912 pdfRangeForPageInSelection(p, p == startPage, p == endPage, startPoint, endPoint);
913 if (!selection.text().trimmed().isEmpty()) {
914 hasAnyText = true;
915 break;
916 }
917 }
918 // While dragging, QPdf can return empty for a frame (geometry hitches, etc.). Latch: once
919 // the range had text, keep m_hasTextSelection true until release; then we finalize.
920 if (m_selectingText) {
921 m_hasTextSelection = m_hasTextSelection || hasAnyText;
922 } else {
923 m_hasTextSelection = hasAnyText;
924 }
925 rebuildTextSelectionHighlights();
926}

931void PdfGraphicsView::updateCurrentPageFromViewport()
932{

934 if (m_currentPage != 0) {
935 m_currentPage = 0;
936 emit currentPageChanged(m_currentPage);
937 }
938 return;
939 }
940 const QPointF sceneCenter = mapToScene(viewport()->rect().center());

945 const qreal d = qAbs(sceneCenter.y() - cy);
946 if (d < bestDist) {
947 bestDist = d;
948 best = i;
949 }
950 }
951 if (best != m_currentPage) {
952 m_currentPage = best;
953 emit currentPageChanged(m_currentPage);
