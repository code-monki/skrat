238/**
239 * @brief Renders a small chevron pixmap for toolbar navigation when theme icons are missing.
240 * @param[in] forward @c true for forward/next direction, @c false for backward/prev.
241 * @param[in] doubled @c true to draw a double chevron (skip/first-last style).
242 * @return QIcon wrapping a transparent 22×22 pixmap with antialiased strokes.
243 */
244QIcon chevronFallbackIcon(bool forward, bool doubled)
245{
246 QPixmap pm(22, 22);
247 pm.fill(Qt::transparent);
248 QPainter p(&pm);
249 p.setRenderHint(QPainter::Antialiasing, true);
250 QPen pen(QApplication::palette().buttonText().color());
251 pen.setWidth(2);
252 pen.setCapStyle(Qt::RoundCap);
253 pen.setJoinStyle(Qt::RoundJoin);
254 p.setPen(pen);
255
256 const int cy = 11;
257 auto drawChevron = [&](int x) {
258 if (forward) {
259 p.drawLine(x - 3, cy - 5, x + 2, cy);
260 p.drawLine(x + 2, cy, x - 3, cy + 5);
261 } else {
262 p.drawLine(x + 3, cy - 5, x - 2, cy);
263 p.drawLine(x - 2, cy, x + 3, cy + 5);
264 }
265 };
266
267 if (doubled) {
268 drawChevron(forward ? 8 : 14);
269 drawChevron(forward ? 14 : 8);
270 } else {
271 drawChevron(11);
272 }
273 return QIcon(pm);
274}

313/**
314 * @brief Strips inline background colors from HTML to avoid pasting unreadable fills.
315 * @param[in] html Clipboard HTML fragment that may contain @c style or legacy bgcolor attributes.
316 * @return Modified HTML with background-related declarations removed.
317 */
318QString stripBackgroundStyles(QString html)
319{
320 // Remove inline background styling while keeping other formatting.
321 html.remove(QRegularExpression(
322 QStringLiteral(R"((?:^|;)\s*background(?:-color)?\s*:\s*[^;\"']+\s*;?)"),

326 QRegularExpression::CaseInsensitiveOption));
327 return html;

376 * @param[in] pageLabel Label showing “Page X of Y”; receives transparent-for-mouse-events.
377 * @param[in] parent Optional parent for lifetime; usually the preview pane widget.
378 */
379 explicit PdfDocumentRibbon(QToolBar *toolBar, QLabel *pageLabel, QWidget *parent = nullptr)
380 : QFrame(parent)
381 , m_toolBar(toolBar)
382 , m_pageLabel(pageLabel)
383 {
384 setFrameShape(QFrame::NoFrame);
385 setObjectName(QStringLiteral("pdfDocumentRibbon"));
386 if (m_toolBar) {
387 m_toolBar->setParent(this);

389 if (m_pageLabel) {
390 m_pageLabel->setParent(this);
391 m_pageLabel->setAlignment(Qt::AlignCenter);
392 m_pageLabel->setAttribute(Qt::WA_TransparentForMouseEvents, true);
393 }
394 setSizePolicy(QSizePolicy::Expanding, QSizePolicy::Fixed);
395 }
396
397 /** @brief Re-runs internal layout after label text or toolbar content changes size. */
398 void recenter() { doLayout(); }
399
400protected:
401 /**
402 * @brief Repositions children when the ribbon’s width or height changes.
403 * @param[in] event Qt resize event forwarded to QFrame.
404 */
405 void resizeEvent(QResizeEvent *event) override
406 {
407 QFrame::resizeEvent(event);
408 doLayout();
409 }
410
411 /**
412 * @brief Returns a preferred height based on toolbar and label size hints.
413 * @return Size hint with unconstrained width and at least 34px height.
414 */
415 QSize sizeHint() const override
416 {
417 int h = 34;
418 if (m_toolBar) {
419 h = qMax(h, m_toolBar->sizeHint().height() + 6);
420 }
421 if (m_pageLabel) {
422 h = qMax(h, m_pageLabel->sizeHint().height() + 6);
423 }
424 return QSize(0, h);
425 }
426
427private:
428 /**
429 * @brief Places the tool bar flush left (vertically centered) and centers the page label.
430 */
431 void doLayout()
432 {
433 if (!m_toolBar || !m_pageLabel) {
434 return;
435 }
436 const int rowH = qMax(1, height());
437 m_toolBar->adjustSize();
438 const int tw = m_toolBar->sizeHint().width();
439 const int th = m_toolBar->sizeHint().height();
440 const int ty = qMax(0, (rowH - th) / 2);
441 m_toolBar->setGeometry(0, ty, tw, th);
442
443 m_pageLabel->adjustSize();
444 const int lw = m_pageLabel->sizeHint().width();
445 const int lh = m_pageLabel->sizeHint().height();
446 const int lx = qMax(0, (width() - lw) / 2);
447 const int ly = qMax(0, (rowH - lh) / 2);
448 m_pageLabel->setGeometry(lx, ly, lw, lh);
449 m_pageLabel->raise();
450 }
451
452 QToolBar *m_toolBar = nullptr;
453 QLabel *m_pageLabel = nullptr;
454};
